\begin{align}
\vec{v} = -\myvec{12\\0}
\end{align}

The resulting rain velocity is 
\begin{align}
\vec{u}+\vec{v} = \myvec{-12\\35}
\end{align}
The desired angle is 
\begin{align}
-\tan^{-1}\phase{\vec{u}+\vec{v}} &= \tan^{-1}\frac{12}{35}
\\
&\approx 20.05 \degree
\end{align}
